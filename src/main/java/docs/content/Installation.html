Using XIS with Spring

XIS currently uses Gradle for build configuration with Spring Boot. Maven support is planned for future releases. Alternatively, a standalone deployment option using xis-boot is available, and Micronaut integration is under consideration. 



plugins {
    id 'one.xis.plugin' version '${xisVersion}'
}

dependencies {

    // xis-libraries
    implementation project(':xis-spring')
    implementation project(':xis-test')

    // Lombok (optional, but recommended, frequently used in examples)
    compileOnly "org.projectlombok:lombok:${lombokVersion}"
    annotationProcessor "org.projectlombok:lombok:${lombokVersion}"

    // Spring Boot
    implementation "org.springframework.boot:spring-boot-starter-web:${springBootVersion}"

    // JUnit 5 for testing
    testImplementation "org.junit.jupiter:junit-jupiter-api:${junitVersion}"
    testRuntimeOnly "org.junit.jupiter:junit-jupiter-engine:${junitVersion}"
}

test {
    useJUnitPlatform()
}
Do not forget to add the xis plugin one.xis.plugin to your build.gradle file.
